# Изменение яркости
```c#
using System;
using System.Management;

BrightnessControl.SetBrightness(70); // Установить 50%
public static class BrightnessControl
{
    /// <param name="brightness">Значение от 0 до 100</param>
    public static void SetBrightness(byte brightness)
    {
        // Подключаемся к пространству имен WMI для мониторов
        ManagementScope scope = new ManagementScope(@"\\.\root\WMI");
        SelectQuery query = new SelectQuery("WmiMonitorBrightnessMethods");

        using (ManagementObjectSearcher searcher = new ManagementObjectSearcher(scope, query))
        {
            using (ManagementObjectCollection instances = searcher.Get())
            {
                foreach (ManagementObject instance in instances)
                {
                    // Вызываем метод WmiSetBrightness
                    // Первый аргумент: таймаут (обычно 1 секунда)
                    // Второй аргумент: уровень яркости (0-100)
                    instance.InvokeMethod("WmiSetBrightness", new object[] { 1, brightness });
                }
            }
        }
    }
}

# Получение информации на удаленном ПК
```C#
using System;
using System.Management;

class RemoteWMI
{
    static void Main()
    {
        string remoteComputer = "192.168.1.100"; // IP или имя ПК
        string domain = "WORKGROUP";             // Домен или имя ПК для локальной учетки
        string username = "Admin";               // Логин
        string password = "YourPassword";       // Пароль

        ConnectionOptions options = new ConnectionOptions();
        options.Username = $"{domain}\\{username}";
        options.Password = password;
        options.Impersonation = ImpersonationLevel.Impersonate; // Уровень олицетворения
        options.EnablePrivileges = true;

        // Формируем путь к пространству имен CIMV2 на удаленном ПК
        ManagementScope scope = new ManagementScope($@"\\{remoteComputer}\root\cimv2", options);

        try
        {
            scope.Connect(); // Устанавливаем соединение
            Console.WriteLine("Соединение установлено!");

            // 1. Информация об ОС
            GetInfo(scope, "SELECT * FROM Win32_OperatingSystem", "Caption", "Version");

            // 2. Информация о процессоре
            GetInfo(scope, "SELECT * FROM Win32_Processor", "Name", "NumberOfCores");

            // 3. Оперативная память (в байтах)
            GetInfo(scope, "SELECT * FROM Win32_ComputerSystem", "TotalPhysicalMemory");
        }
        catch (Exception ex)
        {
            Console.WriteLine("Ошибка подключения: " + ex.Message);
        }
    }

    static void GetInfo(ManagementScope scope, string queryString, params string[] properties)
    {
        ObjectQuery query = new ObjectQuery(queryString);
        using (ManagementObjectSearcher searcher = new ManagementObjectSearcher(scope, query))
        {
            foreach (ManagementObject obj in searcher.Get())
            {
                foreach (var prop in properties)
                {
                    Console.WriteLine($"{prop}: {obj[prop]}");
                }
            }
        }
    }
}
